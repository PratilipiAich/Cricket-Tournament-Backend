## Create player table:

CREATE TABLE player (
playerid uuid, 
first_name text, 
last_name text, 
player_initials text,
player_dob date, 
gender text, 
category text,
nationality text,
player_role text,
player_batting_style text,
player_bowling_style text,
player_status text,
stateteamid uuid,
nationalteamid uuid,

UPDATE series SET host_country = host_country - ['UNKWN']
